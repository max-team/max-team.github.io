# 快速开始

# 安装和使用

# 安装 mars-cli
npm install -g @marsjs/cli

# 创建项目
mars create [projectName]

# 可选项：
# 是否支持 H5，如需支持 H5 请选择：小程序和 H5
? 选择创建项目类型： (Use arrow keys)
❯ 小程序和 H5
  仅小程序

# 开发模式
mars serve [--target, -t swan (默认) | wx | h5]

# 构建模式
mars build [--target, -t swan (默认) | wx | h5]
